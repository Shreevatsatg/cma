var canvas=document.getElementById("canvas"); var ctx=canvas.getContext("2d"); var isDrawing=false; var color="#000000"; var thickness=1; ctx.strokeStyle=color; ctx.lineWidth=thickness; document.getElementById("color").addEventListener("input",function(event){ color=event.target.value; ctx.strokeStyle=color; }); document.getElementById("thickness").addEventListener("input",function(event){ thickness=event.target.value; ctx.lineWidth=thickness; }); canvas.addEventListener("mousedown",function(event){ isDrawing=true; ctx.beginPath(); ctx.moveTo(event.pageX-canvas.offsetLeft,event.pageY-canvas.offsetTop); }); canvas.addEventListener("mousemove",function(event){ if(isDrawing){ ctx.lineTo(event.pageX-canvas.offsetLeft,event.pageY-canvas.offsetTop); ctx.stroke(); } }); canvas.addEventListener("mouseup",function(){ isDrawing=false; }); document.getElementById("clear").addEventListener("click",function(){ ctx.clearRect(0,0,canvas.width,canvas.height); });
